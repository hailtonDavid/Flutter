-------------------------Exemplo 01

import "package:flutter/material.dart";

void main() => runApp(MyApp());

class MyApp extends StatelessWidget {
  handleButtonClick() {
    print("Testando 1,2 3");
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
        title: "titulo de exemplo",
        home: Scaffold(
            appBar: AppBar(
              title: Text("Titulo de exemplo"),
            ),
            body: Column(
                mainAxisAlignment: MainAxisAlignment.center,
                children: <Widget>[
                  Text('Olá mundo'),
                  FlatButton(
                      child: Text("Me aperte"), onPressed: handleButtonClick)
                ])));
  }
}

-------------------------Exemplo 06

import "package:flutter/material.dart";

void main() => runApp(MyApp());

class MyApp extends StatelessWidget {
  var lista = [
    'Hailton',
    'Neusa',
    'Sophia',
    'Vitor Hugo',
    'Maria Eduarda',
    'Fulana'
  ];

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
        title: "titulo de exemplo",
        home: Scaffold(
            appBar: AppBar(
              title: Text("Titulo de exemplo"),
            ),
            body: ListView.builder(
              itemCount: lista.length,
              itemBuilder: (context, int index) {
                var textStyle = TextStyle(fontSize: 20, color: Colors.red);
                return Text(lista[index], style: textStyle);
              },
            )));
  }
}
